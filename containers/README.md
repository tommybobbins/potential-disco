````

````
type AutoGenerated struct {
	Data Data `json:"data"`
}
type Generationmix struct {
	Fuel string  `json:"fuel"`
	Perc float64 `json:"perc"`
}
type Data struct {
	From          string          `json:"from"`
	To            string          `json:"to"`
	Generationmix []Generationmix `json:"generationmix"`
}
